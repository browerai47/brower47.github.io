5th Period Java Project: GUI Calculator

This page features the source code for our working graphical calculator, built using Java's Swing library.

Java Source Code: CalculatorShell.java


import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import javax.sound.sampled.AudioInputStream;
import javax.sound.sampled.AudioSystem;
import javax.sound.sampled.Clip;
import java.io.File;
import java.net.URL;

public class FlappyBird extends JPanel implements ActionListener, KeyListener {

    public static final int WIDTH = 800;
    public static final int HEIGHT = 600;
    
    private Timer timer;
    private boolean gameOver = false;
    private int score = 0;
    // --- NEW: Reference to the audio clip ---
    private Clip gameMusicClip; 

    private Rectangle bird;
    private int yVelocity = 0;
    private final int GRAVITY = 1;
    private final int JUMP_STRENGTH = -12;

    private Rectangle pipe1Top, pipe1Bottom;
    private Rectangle pipe2Top, pipe2Bottom;
    private final int PIPE_WIDTH = 60;
    private final int PIPE_GAP = 150;
    private int PIPE_SPEED = 5;

    public FlappyBird() {
        pipe1Top = new Rectangle(WIDTH, 0, PIPE_WIDTH, 200);
        pipe1Bottom = new Rectangle(WIDTH, 200 + PIPE_GAP, PIPE_WIDTH, HEIGHT - (200 + PIPE_GAP) - 100);
        pipe2Top = new Rectangle(WIDTH + WIDTH/2, 0, PIPE_WIDTH, 300);
        pipe2Bottom = new Rectangle(WIDTH + WIDTH/2, 300 + PIPE_GAP, PIPE_WIDTH, HEIGHT - (300 + PIPE_GAP) - 100);
        bird = new Rectangle(100, HEIGHT/2, 30, 30);
        this.setPreferredSize(new Dimension(WIDTH, HEIGHT));
        this.setBackground(Color.CYAN);
        this.setFocusable(true);
        this.addKeyListener(this);

        timer = new Timer(20, this);
        timer.start();

        playMusic("Bad Piggies theme 1 hour.wav");
    }

    public void playMusic(String musicFileName) {
        new Thread(new Runnable() {
            public void run() {
                try {
                    URL soundUrl = getClass().getClassLoader().getResource(musicFileName);
                    if (soundUrl == null) {
                        File file = new File(musicFileName);
                        if (!file.exists()) {
                             System.out.println("Music file not found: " + musicFileName);
                             return;
                        }
                        soundUrl = file.toURI().toURL();
                    }
                    
                    AudioInputStream audioInput = AudioSystem.getAudioInputStream(soundUrl);
                    // Assign the created clip to the global variable
                    gameMusicClip = AudioSystem.getClip(); 
                    gameMusicClip.open(audioInput);
                    gameMusicClip.loop(Clip.LOOP_CONTINUOUSLY);
                    gameMusicClip.start();
                } catch (Exception ex) {
                    System.out.println("Error playing music:");
                    ex.printStackTrace();
                }
            }
        }).start();
    }
    
    // --- NEW Method: Stop the music playback ---
    public void stopMusic() {
        if (gameMusicClip != null && gameMusicClip.isRunning()) {
            gameMusicClip.stop();
        }
    }

    // --- NEW Method: Restart the music playback (for restarting the game) ---
    public void restartMusic() {
        if (gameMusicClip != null) {
            gameMusicClip.setFramePosition(0); // Rewind to the beginning
            gameMusicClip.loop(Clip.LOOP_CONTINUOUSLY);
            gameMusicClip.start();
        } else {
            // If it somehow wasn't loaded, try loading it again
            playMusic("Bad Piggies theme 1 hour.wav");
        }
    }
    
    @Override
    public void actionPerformed(ActionEvent e) {
        update();
        repaint();
    }
  
    public void update() {
        // ...
        // Check Collisions
        if (bird.intersects(pipe1Top) || bird.intersects(pipe1Bottom) ||
            bird.intersects(pipe2Top) || bird.intersects(pipe2Bottom)) {
            if (!gameOver) { // Check if this is the first time we detect game over
                stopMusic(); // Stop music when dead
            }
            gameOver = true;
            PIPE_SPEED = 0;
        }

        if (!gameOver) {
            // ... (rest of the update logic) ...
            yVelocity += GRAVITY;
            bird.y += yVelocity;
            PIPE_SPEED = 5;
            
            if (bird.y > HEIGHT - 120 || bird.y < 0) {
                bird.y = (bird.y > HEIGHT - 120) ? HEIGHT - 120 : 0;
                if (!gameOver) { // Check if this is the first time we detect game over
                    stopMusic(); // Stop music when hitting floor/ceiling
                }
                gameOver = true;
                PIPE_SPEED = 0;
            }
        }
        // ... (Pipe movement/recycling logic remains the same) ...
        pipe1Top.x -= PIPE_SPEED;
        pipe1Bottom.x -= PIPE_SPEED;
        pipe2Top.x -= PIPE_SPEED;
        pipe2Bottom.x -= PIPE_SPEED;

        // Recycle Pipes
        if (pipe1Top.x + PIPE_WIDTH < 0) {
            pipe1Top.x = WIDTH;
            pipe1Bottom.x = WIDTH;
            score++;
            int newHeight = (int)(Math.random() * 300) + 50;
            pipe1Top.height = newHeight;
            pipe1Bottom.y = newHeight + PIPE_GAP;
            pipe1Bottom.height = HEIGHT - (newHeight + PIPE_GAP) - 100;
        }
        if (pipe2Top.x + PIPE_WIDTH < 0) {
            pipe2Top.x = WIDTH;
            pipe2Bottom.x = WIDTH;
            score++;
            int newHeight = (int)(Math.random() * 300) + 50;
            pipe2Top.height = newHeight;
            pipe2Bottom.y = newHeight + PIPE_GAP;
            pipe2Bottom.height = HEIGHT - (newHeight + PIPE_GAP) - 100;
        }
    }

    @Override
    public void paintComponent(Graphics g) {
        super.paintComponent(g);
        // ... (Drawing logic remains the same) ...
        g.setColor(Color.ORANGE);
        g.fillRect(0, HEIGHT - 100, WIDTH, 100);
        g.setColor(Color.GREEN);
        g.fillRect(0, HEIGHT - 100, WIDTH, 20);
        
        g.setColor(Color.WHITE);
        g.setFont(new Font("Arial", Font.BOLD, 50));
        g.drawString("Score: " + score, WIDTH/2 - 20, 50);

        g.setColor(Color.YELLOW);
        g.fillRect(bird.x, bird.y, bird.width, bird.height);

        g.setColor(Color.GREEN.darker());
        g.fillRect(pipe1Top.x, pipe1Top.y, pipe1Top.width, pipe1Top.height);
        g.fillRect(pipe1Bottom.x, pipe1Bottom.y, pipe1Bottom.width, pipe1Bottom.height);
        
        g.fillRect(pipe2Top.x, pipe2Top.y, pipe2Top.width, pipe2Top.height);
        g.fillRect(pipe2Bottom.x, pipe2Bottom.y, pipe2Bottom.width, pipe2Bottom.height);

        if (gameOver) {
            g.setColor(Color.RED);
            g.drawString("Game Over!", WIDTH/2 - 150, HEIGHT/2);
            g.setFont(new Font("Arial", Font.PLAIN, 20));
            g.drawString("Press Space to Restart", WIDTH/2 - 100, HEIGHT/2 + 50);
        }
    }

    // Keyboard Stubs
    @Override
    public void keyTyped(KeyEvent e) {}
    @Override
    public void keyReleased(KeyEvent e) {}
    @Override
    public void keyPressed(KeyEvent e) {
        if (e.getKeyCode() == KeyEvent.VK_SPACE) {
            if (gameOver) {
                // Reset Game State
                bird.y = HEIGHT/2;
                yVelocity = 0;
                pipe1Top.x = WIDTH;
                pipe1Bottom.x = WIDTH;
                pipe2Top.x = WIDTH + WIDTH/2;
                pipe2Bottom.x = WIDTH + WIDTH/2;
                score = 0;
                gameOver = false;
                // --- NEW: Restart the music when the game restarts ---
                restartMusic(); 
            } else {
                // Jump!
                yVelocity = JUMP_STRENGTH;
            }
        }
    }

    public static void main(String[] args) {
        JFrame frame = new JFrame("Flappy Bird");
        FlappyBird game = new FlappyBird();
        frame.add(game);
        frame.pack();
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
        frame.setResizable(false);
    }
}

Project Logic Explanation

The Java program simulates a basic European roulette game where a user can bet on a single number, a color (Red/Black), or Odd/Even numbers.

  • The `main` function manages the game loop and player balance.
  • `spinWheel()` uses a `Random` number generator to determine the winning number (0-36).
  • `checkWin()` compares the user's bet type and choice with the result of the spin and returns a boolean.
  • The code handles user input validation and calculates winnings based on typical roulette odds (35:1 for number, 1:1 for even chances).
